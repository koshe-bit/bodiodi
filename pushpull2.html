function buildGrid(){ nodes = []; rowMap = []; const cx = W * 0.50; const cy = H * 0.52; const orbW = W * 0.52; const orbH = H * 0.46; const RINGS = 12; const centerNode = { r: 0, c: 0, rowCols: 1, baseX: cx, baseY: cy, x: cx, y: cy, seed: Math.random() * 1000 }; nodes.push(centerNode); rowMap.push([centerNode]); for (let ring = 1; ring < RINGS; ring++){ const t = ring / (RINGS - 1); const easeT = Math.pow(t, 0.85); const rx = orbW * easeT; const ry = orbH * easeT; const opacity = 1 - Math.pow(t, 1.6) * 0.72; const numPts = Math.round(6 + ring * 2.8); const row = []; for (let i = 0; i < numPts; i++){ const angle = (i / numPts) * Math.PI * 2; const nx = cx + Math.cos(angle) * rx; const ny = cy + Math.sin(angle) * ry; const node = { r: ring, c: i, rowCols: numPts, baseX: nx, baseY: ny, x: nx, y: ny, seed: Math.random() * 1000, ringOpacity: opacity }; row.push(node); nodes.push(node); } rowMap.push(row); } }
